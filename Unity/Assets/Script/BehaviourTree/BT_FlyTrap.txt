tree "Alive"
    fallback
        tree "Attack"
        tree "Hide"

tree "Attack"
    sequence
        IsAlert
        not IsTimer 0
        Wait 1
        SetSprite 1
        SetColor 0
        SetColliders 1
        SetAttack 0 1
        SetTimer 0 2
        UnsetAlert

tree "Hide"
    sequence
        SetSprite 0
        SetColor 1
        SetColliders 0
        SetAttack 0 0
